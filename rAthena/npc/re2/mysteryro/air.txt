dali02,142,81,3	script	Dimensional Device#sara1	10007,{
	set .@airassault_time,checkquest(15050,PLAYTIME);
		if (.@airassault_time == 2) {
			erasequest 15050;
			erasequest 15051;
			end;
		}
		else  if ((.@airassault_time == 1) || (.@airassault_time == 0)) {
			set .@airreentrada,checkquest(15051,PLAYTIME);
				if ((.@airreentrada == -1) || (.@airreentrada == 1) || (.@airreentrada == 0)) {
					set .@party_id,getcharid(1);
					set .@p_name$,getpartyname(.@party_id);
					set .@md_name$,"Airship Assault";
					
					if (!instance_check_party(.@party_id,1)) {
						mes "You must create a party!";
						close;
					}
					if (getcharid(0) == getpartyleader(.@party_id,2))
						switch(select("Power Up the Device","Use the Device","Cancel")) 
						{
							case 1:
								if (instance_create(.@md_name$) < 0) {
									mes "Party: "+.@p_name$;
									mes "Party Leader: "+strcharinfo(0);
									mes "^0000ff"+.@md_name$+" ^000000- Reservation failed!";
									close;

									mes "^ff0000Use it again to enter^000000";
									close;
								}

								mes "^ff0000Re-entry is not possible within 23 hours after the first entry.^000000";
								next;
								switch(instance_enter(.@md_name$))
								{
									case 0:
										mapannounce "dali02",strcharinfo(0)+", member of the party "+.@p_name$+" entered the instance "+.@md_name$+".",bc_map,"0x00ff99";
										specialeffect2 EF_WARP;
										end;
									case 1:
										mes "Only the registered members can enter the instance "+.@md_name$+".";
										close;
									case 2:
										mes "The memorial dungeon "+.@md_name$+" does not exist.";
										mes "The party leader did not generate the dungeon yet.";
										close;
									case 3:
										mes "Unkwnon error";
										close;
								}
							case 3:
								close;
						}
				} else {
					mes "^ff0000This memorial dungeon will not be able to re-enter within 23 hours of the last entry.^000000";
					close;
				}
		}
		else if (.@airassault_time == -1) {
			set .@party_id,getcharid(1);
			set .@p_name$,getpartyname(.@party_id);
			set .@md_name$,"Airship Assault";
			
			if (!instance_check_party(.@party_id,1)) {
				mes "You must create a party!";
				close;
			}
			if (getcharid(0) == getpartyleader(.@party_id,2)) {
				switch(select("Power Up the Device","Use the Device","Cancel"))
				{
					case 1:
						if (instance_create(.@md_name$) < 0) {
							mes "Party: "+.@p_name$;
							mes "Party Leader: "+strcharinfo(0);
							mes "^0000ff"+.@md_name$+" ^000000- Reservation failed!";
							close;
						} else {
							mes "^ff0000After the dimensional device activities^000000";
							mes "^ff0000Use it again to enter^000000";

						next;

						{
							case 0:
